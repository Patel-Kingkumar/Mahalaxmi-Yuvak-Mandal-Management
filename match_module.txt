-- View Player Stats with Match & User Info (Recommended)
SELECT 
    ps.PlayerStatId,
    m.MatchId,
    u.FullName  AS PlayerName,
    ps.TeamName,
    ps.RunsScored,
    ps.BallsPlayed,
    ps.Fours,
    ps.Sixes,
    ps.WicketsTaken,
    ps.OversBowled,
    ps.RunsGiven,
    ps.Catches
FROM PlayerStats ps
INNER JOIN Matches m ON ps.MatchId = m.MatchId
INNER JOIN Users u ON ps.UserId = u.Id;
